快速安装jdk

2020-02-04 1 min read 
curl -L -b "oraclelicense=a" -O https://download.oracle.com/otn-pub/java/jdk/8u201-b09/42970487e3af4f5aa5bca3f542482c60/jdk-8u201-linux-x64.tar.gz
sudo mkdir /usr/local/oracle-java-8
sudo tar -zxf jdk-8u201-linux-x64.tar.gz -C /usr/local/oracle-java-8

sudo update-alternatives --install "/usr/bin/java" "java" "/usr/local/oracle-java-8/jdk1.8.0_201/bin/java" 1500
sudo update-alternatives --install "/usr/bin/javac" "javac" "/home/jdk1.8.0_131/bin/javac" 1500
sudo update-alternatives --install "/usr/bin/javaws" "javaws" "/home/jdk1.8.0_131/bin/javaws" 1500
sudo update-alternatives --install "/usr/bin/jcmd" "jcmd" "/home/jdk1.8.0_131/bin/jcmd" 1500
